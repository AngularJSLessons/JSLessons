OOP - Prototype (alternative class), Constructor, Encapsulation

! - Каждая функция имеет свойство prototype, представляющее прототип функции. То есть свойство User.prototype представляет прототип объектов User. И любые свойства и методы, которые будут определены в User.prototype, будут общими для всех объектов User.

User.prototype.age = 4; // all User objects have age property with 4 value.

Constructor - function

Example

function User(pName, pAge) {
    this.name = pName;
    this.age = pAge;
    this.displayInfo = function(){
        document.write("Имя: " + this.name + "; возраст: " + this.age);
    };
};
var tom = new User("Том", 26);
tom.name=34;
console.log(tom.name);

Encapsulation

Example

function User(pName, pAge) {
    var _name = pName; // private in JS
    this.age = pAge;
    this.getName = function(){
        return _name;
    }
    this.displayInfo = function(){
        document.write("Имя: " + this.name + "; возраст: " + this.age);
    };
};
var tom = new User("Том", 26);
console.log(tom._name); //undefined
console.log(tom.getName); //Tom

Inheritance

// base function constructor
function User (name, age) {
    this.name = name;
    this.age = age;
    this.go = function(){document.write(this.name + " идет 
");}
    this.displayInfo = function(){
        document.write("Имя: " + this.name + "; возраст: " + this.age + "
");
    };
}
User.prototype.maxage = 110;

// child class constructor
function Employee(name, age, comp){
    User.call(this, name, age); // call base function constructor
    this.company = comp;
    this.displayInfo = function(){
        document.write("Имя: " + this.name + "; возраст: " + this.age + "; компания: " + this.company + "
");
    };
}
Employee.prototype = Object.create(User.prototype); // Employee extends user
